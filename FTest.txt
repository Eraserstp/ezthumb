Functional Test Plan
====================

Test File:
small.wmv: 4 i-frames, 10 seconds, backward seekable
middle.avi: 40 minutes video clip
broken.avi: a broken video clip
jumppy.avi: composed by 3 clips each has minutes offset (cat broken.avi xag xap > jumppy.avi)
bumppy.avi: content the rewound time stamp (cat broken.avi xar xaf > bummpy.avi)
DejaVuSans.ttf: TTF font file


#001 Generate 4x4 thumbnails, 50% screen size
-    More explain lines should start with a '-'
ezthumb small.wmv

#002 Generate 4x4 thumbnails, 50% screen size and accurate frames
ezthumb --accurate small.wmv

#003 Display media quick list
ezthumb -i *

#004 Display detailed media information
ezthumb -I small.wmv

#005 Do a full scan and display the media information
ezthumb -I -d scan small.wmv

#006 Do a quick scan and display the media information
ezthumb -I -d fast small.wmv

#007 Specify the canvas colour: Media information is BLUE, Timestamp is RED,
-    The background is grey
ezthumb -c 0000ff:ff0000:7f7f7f small.wmv

#008 Only setup the background colour:
ezthumb -c ::7f7f7f small.wmv

#009 Check the result of '-d' option: see the different of below
ezthumb -i broken.avi
ezthumb -i -d fast broken.avi

#010 Set up the thumbnail edge
ezthumb -e 4 small.wmv


#021 Test the internal font with different size:
ezthumb -F 16:8 small.wmv

#022 Test the internal font with different size and colour:
ezthumb -F 16:8 -c 00ff00:0000ff small.wmv

#023 Test the external font with DejaVuSans.ttf:
ezthumb -f ./DejaVuSans.ttf small.wmv

#024 Use the external font DejaVuSans.ttf and change its size
ezthumb -f ./DejaVuSans.ttf -F 16:8 small.wmv

#025 Use the same font but only change the size and colour of the time stamp:
ezthumb -f ./DejaVuSans.ttf -F :16 -c :ff0000: small.wmv


#030 Set up an 8x1 grid
ezthumb -g 8x1 small.wmv

#031 Set up a 1x8 grid
ezthumb -g 1x8 small.wmv

#032 No grid and canvas at all, save to separate 8 screen shots:
ezthumb -g 0x8 small.wmv

#033 No grid and canvas, save all i-frame separately:
ezthumb -g 0x0 small.wmv

#034 Take snapshots and change to expected size in percentage:
ezthumb -g 3x5 -s 40% small.wmv
ezthumb -g 3x5 -s 20% small.wmv

#035 Take snapshots and change to expected size in pixel:
ezthumb -g 3x5 -s 300x100 small.wmv

#036 Limit the width of the canvas and automatically fit the snapshots
ezthumb -g 3x5 -s 300x100 -w 1024 small.wmv

#037 Set up the gaps between screen shots in pixel:
ezthumb -g 3x5 --gap-shots 0 small.wmv
ezthumb -g 3x5 --gap-shots 10 small.wmv

#038 Set up the gaps between screen shots in proportion:
ezthumb -g 3x5 --gap-shots 0% small.wmv
ezthumb -g 3x5 --gap-shots 10% small.wmv

#039 Set up the canvas margin in pixel:
ezthumb -g 3x5 --gap-margin 0 small.wmv
ezthumb -g 3x5 --gap-margin 20 small.wmv

#040 Set up the canvas margin in proportion:
ezthumb -g 3x5 --gap-margin 0% small.wmv
ezthumb -g 3x5 --gap-margin 20% small.wmv

#041 Turn off the media information
ezthumb --opt-info off small.wmv

#042 Place the media information to the right corner
ezthumb --pos-info rt small.wmv
ezthumb --pos-info rc small.wmv
ezthumb --pos-info rb small.wmv

#043 Place the media information to the center
ezthumb --pos-info mt small.wmv
ezthumb --pos-info mc small.wmv
ezthumb --pos-info mb small.wmv

#044 Turn off the timestamp
ezthumb --opt-time off small.wmv

#045 Place the timestamp to the left-top, left-bottom and right-bottom corner:
ezthumb --pos-time lt small.wmv
ezthumb --pos-time lb small.wmv
ezthumb --pos-time rb small.wmv

#046 Place the timestamp to the unusual place:
ezthumb --pos-time lc small.wmv
ezthumb --pos-time mt small.wmv
ezthumb --pos-time mc small.wmv
ezthumb --pos-time mb small.wmv
ezthumb --pos-time rc small.wmv

#047 Make sure the 'tt' place does NOT work for media information and timestamp
ezthumb --pos-info tt small.wmv
ezthumb --pos-time tt small.wmv


#060 Generate JPG outputs with the different qualities. 
-    The quality over 100 should be ignored.
ezthumb -m jpg@20 small.wmv
ezthumb -m jpg@100 small.wmv
ezthumb -m jpg@120 small.wmv

#061 Generate PNG outputs. The tailing '@20' should be ignored.
ezthumb -m png small.wmv
ezthumb -m png@20 small.wmv

#062 Generate GIF outputs. 
ezthumb -m gif small.wmv

#063 Generate the animation GIF outputs. The time interval is 500 milliseconds:
ezthumb -m gif@500 small.wmv

#064 Generate the animation GIF outputs without timestamps:
ezthumb -m gif@300 --opt-time off small.wmv

#065 Generate the animation GIF outputs by key-frames:
ezthumb -m gif@1000 -g 0 small.wmv
